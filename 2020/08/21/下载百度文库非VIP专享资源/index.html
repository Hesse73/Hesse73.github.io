PPT文档

百度文库的PPT是以图片显示的，如果下载只能把图片全部爬取然后转格式

下面用Selenium + Chrome来操作一下：

import os
import time
import requests
from selenium import webdriver
from selenium.webdriver.common.keys import Keys

#设置链接地址
url = 'https://wenku.baidu.com/view/d36080e386c24028915f804d2b160b4e777f81ee.html'

#使用Chrome打开
driver = webdriver.Chrome()
# 用get打开百度文库页面
driver.get(url)
#找到“继续浏览”的按钮
button = driver.find_element_by_class_name("down-arrow.goBtn")
#此时网页会自动下拉到按钮位置，但由于底栏浮动在表面，无法点击
"""
#方法一：隐藏底栏
#js = 'document.getElementsByClassName("reader-tools-bar-wrap tools-bar-small tools-bar-smaller")[0].setAttribute("style","display:none;")'
#driver.execute_script(js)
方法二：滚动页面至按钮可见
"""
js = 'document.getElementsByClassName("down-arrow goBtn")[0].scrollIntoView()'
driver.execute_script(js)
#点击按钮
button.click()
#获取页面数量
counts = len(driver.find_elements_by_css_selector(
    'div[class^="ppt-page-item"]'))
#循环获取图片并保存
i = 0
while(i < counts):
    try:
        #获取第i个界面
        page_class = "ppt-image-wrap"
        page = driver.find_elements_by_class_name(page_class)[i]
        #滚动到页面位置
        driver.execute_script(
            "document.getElementsByClassName('ppt-image-wrap')["+str(i)+"].scrollIntoView();")
        #获取图片地址并下载保存
        pic_url = page.find_element_by_tag_name('img').get_attribute('src')
        pic = requests.get(pic_url)
        #保存图片到‘save’文件夹中
        filename = "Pic"+str(i)+".jpg"
        with open('save\\'+filename, "wb") as picfile:
            picfile.write(pic.content)
        pic.close()
        i += 1
        #显示已获取图片数
        os.system('cls')
        print(str(i)+'Picture(s) downloaded')
    except:
        #可能会由于请求太频繁而出错
        time.sleep(1)

# 关闭浏览器
driver.close()
