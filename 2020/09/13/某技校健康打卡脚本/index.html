Slenium版本

很简单，直接模拟登录就好了……

import time
from selenium import webdriver
from selenium.webdriver.common.keys import Keys
from selenium.common.exceptions import NoSuchElementException

#设置链接地址
url = r'https://passport.ustc.edu.cn/login?service=https%3A%2F%2Fweixine.ustc.edu.cn%2F2020%2Fcaslogin'
#设置登录信息
usr = '认证系统用户名'
pwd = '认证系统密码'

#打开chrome
driver = webdriver.Chrome()

# 用get打开登录页面并找到填充的元素
driver.get(url)
#自动登录
InputName = driver.find_element_by_id('username')
InputPwd = driver.find_element_by_id('password')
LoginBtn = driver.find_element_by_id('login')
InputName.send_keys(usr)
InputPwd.send_keys(pwd)
LoginBtn.click()
#此处需要手动暂停，已保证网页自动加载以往打卡配置
time.sleep(1)
#进入上报界面找到上报按钮，点击即可
ReportBtn = driver.find_element_by_id('report-submit-btn')
#滚动至按钮显示
js = 'document.getElementById("report-submit-btn").scrollIntoView()'
driver.execute_script(js)
#点击按钮
ReportBtn.click()
try:
    AlertMsg = driver.find_element_by_class_name('alert')
    print(AlertMsg.text)
except:
    print('failed with clicking report button')
driver.close()
